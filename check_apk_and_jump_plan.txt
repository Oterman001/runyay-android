│ 应用市场跳转更新方案                                                         │
│                                                                              │
│ Context                                                                      │
│                                                                              │
│ 当前更新流程直接从 OSS 下载 APK，耗费服务端流量。若 App                      │
│ 已在华为/小米/OPPO/Vivo/Google Play 上架，优先跳转市场更新可节省流量。       │
│ 需处理"部分市场已上架、部分未上架"的情况，且要兼容 Google Play。             │
│                                                                              │
│ ---                                                                          │
│ 核心设计：服务端返回 Map，客户端按已安装市场匹配                             │
│                                                                              │
│ 为什么用 Map 而不是单个 marketUrl                                            │
│                                                                              │
│ 单个字段只能覆盖一家市场。不同设备（华为/小米/OPPO/Google）需要各自的跳转链  │
│ 接，且各家市场的 URI Scheme 不同。服务端用 Map                               │
│ 分别维护每家市场的上架状态，只有上架了才填写对应                             │
│ key，客户端根据设备已安装的市场做匹配。                                      │
│                                                                              │
│ ---                                                                          │
│ 服务端响应字段设计                                                           │
│                                                                              │
│ // GetLatestVersionResponse.kt 新增                                          │
│ @SerializedName("marketUrls")                                                │

│ 匹配逻辑（客户端）                                                           │
│                                                                              │
│ for each Market in MARKETS (按优先级排列):                                   │
│     if market app is installed on device:                                    │
│         if marketUrls[market.serverKey] exists:                              │
│             → 使用该 URL 跳转，停止搜索                                      │
│         else:                                                                │
│             → 该市场未上架，跳过（不要用兜底 scheme 打开一个找不到 App       │
│ 的市场）                                                                     │
│ → 所有市场均无匹配 → 退化为 APK 下载流程                                     │
│                                                                              │
│ 关键规则：客户端不自行构造 scheme，完全依赖服务端提供的                      │
│ URL。这样能保证"未上架"的市场不会被错误打开（否则用户在市场里找不到 App      │
│ 体验差）。                                                                   │

│     fun resolve(context: Context, marketUrls: Map<String, String>?):         │
│ ResolvedMarket? {                                                            │
│         if (marketUrls.isNullOrEmpty()) return null                          │
│         val pm = context.packageManager                                      │
│         return MARKETS.firstNotNullOfOrNull { market ->                      │
│             val url = marketUrls[market.serverKey] ?:                        │
│ return@firstNotNullOfOrNull null                                             │
│             val installed = runCatching { pm.getPackageInfo(market.pkgName,  │
│ 0) }.isSuccess                                                               │
│             if (installed) ResolvedMarket(market.label, url) else null       │
│         }                                                                    │
│     }                                                                        │
│                                                                              │
│     fun open(context: Context, resolved: ResolvedMarket): Boolean {          │
│         return runCatching {                                                 │
│             val uri = Uri.parse(resolved.url)                                │
│             context.startActivity(                                           │
│                 Intent(Intent.ACTION_VIEW,                                   │
│ uri).addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)                                 │
│             )                                                                │
│             true                                                             │
│         }.getOrDefault(false)                                                │
│     }                                                                        │
│ }                                                                            │
